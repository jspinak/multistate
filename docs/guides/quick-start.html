Quick Start Guide

Get up and running with MultiState in 5 minutes!

Installation​

pip install multistate

Or install from source:

git clone https://github.com/jspinak/multistate.git
cd multistate
pip install -e .

Your First MultiState Application​

Let's build a simple GUI application with multiple panels that can be opened simultaneously.

Step 1: Import and Initialize​

from multistate import StateManager, SearchStrategy

# Create a state manager
manager = StateManager(
    search_strategy=SearchStrategy.DIJKSTRA,  # Use optimal pathfinding
    log_transitions=True  # Enable transition logging
)

Step 2: Define Your States​

# Add application states
main_window = manager.add_state('main', 'Main Window')
toolbar = manager.add_state('toolbar', 'Toolbar')
sidebar = manager.add_state('sidebar', 'Sidebar')
search_panel = manager.add_state('search', 'Search Panel')
properties_panel = manager.add_state('properties', 'Properties Panel')

# Add a modal dialog (blocks other states)
settings_modal = manager.add_state(
    'settings',
    'Settings Dialog',
    blocking=True  # This state occludes others
)

Step 3: Define Transitions​

# Transition that opens multiple UI components at once
manager.add_transition(
    id='open_workspace',
    name='Open Workspace',
    from_states=['main'],
    activate_states=['toolbar', 'sidebar'],  # Activate BOTH
    path_cost=1.0
)

# Transition to show panels
manager.add_transition(
    id='show_panels',
    name='Show All Panels',
    from_states=['toolbar', 'sidebar'],
    activate_states=['search', 'properties'],  # Activate multiple panels
    path_cost=1.5
)

# Transition to open modal
manager.add_transition(
    id='open_settings',
    name='Open Settings',
    from_states=['main'],
    activate_states=['settings'],
    path_cost=0.5
)

Step 4: Multi-Target Pathfinding​

Find the optimal path to reach multiple states:

# Start from main window
manager.activate_states({'main'})

# Find path to reach BOTH search and properties panels
path = manager.find_path_to(['search', 'properties'])

if path:
    print(f"Found path with {len(path.transitions)} steps")
    print(f"Total cost: {path.total_cost}")

    # Show the path
    for i, transition_id in enumerate(path.transitions_sequence):
        transition = manager.get_transition(transition_id)
        print(f"{i+1}. {transition.name}")

Output:

Found path with 2 steps
Total cost: 2.5
1. Open Workspace
2. Show All Panels

Step 5: Execute Transitions with Rollback​

# Execute with automatic rollback on failure
success = manager.execute_transition('open_workspace')

if success:
    print("Workspace opened successfully!")
    print(f"Active states: {manager.get_active_states()}")
else:
    print("Failed to open workspace - state rolled back")

Step 6: Handle Occlusions​

# Open modal (will occlude other states)
manager.activate_states({'settings'})

# Check what's occluded
occluded = manager.get_occluded_states()
print(f"Occluded states: {occluded}")

# Close modal - will generate reveal transition
manager.deactivate_state('settings')
# Previously hidden states are automatically restored!

Complete Example​

Here's everything together in a working example:

from multistate import StateManager, SearchStrategy

def demo_gui_automation():
    """Demonstrate MultiState for GUI automation."""

    # Initialize
    manager = StateManager(SearchStrategy.DIJKSTRA)

    # Define states
    states = {
        'main': 'Main Window',
        'toolbar': 'Toolbar',
        'sidebar': 'Sidebar',
        'content': 'Content Area',
        'search': 'Search Panel',
        'properties': 'Properties Panel',
        'debug': 'Debug Console'
    }

    for state_id, name in states.items():
        manager.add_state(state_id, name)

    # Create state group for UI components
    manager.add_group('workspace', ['toolbar', 'sidebar', 'content'])

    # Define transitions
    transitions = [
        {
            'id': 'startup',
            'from': [],
            'activate': ['main'],
            'cost': 0.1
        },
        {
            'id': 'open_workspace',
            'from': ['main'],
            'activate': ['toolbar', 'sidebar', 'content'],  # Multi-state!
            'cost': 1.0
        },
        {
            'id': 'show_all_panels',
            'from': ['toolbar'],
            'activate': ['search', 'properties', 'debug'],  # Three at once!
            'cost': 2.0
        }
    ]

    for trans in transitions:
        manager.add_transition(
            trans['id'],
            trans['id'].replace('_', ' ').title(),
            trans['from'],
            trans['activate'],
            [],  # No states to exit
            trans['cost']
        )

    # Demonstrate multi-target pathfinding
    print("🎯 Multi-Target Pathfinding Demo")
    print("-" * 40)

    # Start fresh
    manager.clear_active_states()
    manager.activate_states({'main'})

    # Find path to ALL three panels
    targets = ['search', 'properties', 'debug']
    print(f"Finding path to reach ALL: {targets}")

    path = manager.find_path_to(targets)
    if path:
        print(f"✅ Path found: {len(path.transitions)} steps, cost: {path.total_cost:.1f}")

        # Execute the path
        for transition_id in path.transitions_sequence:
            success = manager.execute_transition(transition_id)
            trans = manager.get_transition(transition_id)
            status = "✓" if success else "✗"
            print(f"  {status} {trans.name}")

        # Verify all targets reached
        active = manager.get_active_states()
        reached = all(t in active for t in targets)
        print(f"\nAll targets reached: {'✅ Yes' if reached else '❌ No'}")
        print(f"Active states: {active}")
    else:
        print("❌ No path found")

    # Demonstrate state groups
    print("\n👥 State Groups Demo")
    print("-" * 40)

    # Activate entire workspace at once
    manager.activate_group('workspace')
    print(f"Activated 'workspace' group")
    print(f"Active states: {manager.get_active_states()}")

    # Demonstrate complexity analysis
    print("\n📊 System Complexity")
    print("-" * 40)
    complexity = manager.analyze_complexity()
    print(f"Total states: {complexity['num_states']}")
    print(f"Total transitions: {complexity['num_transitions']}")
    print(f"Reachable states: {complexity['reachable_states']}")
    print(f"Active states: {complexity['active_states']}")

if __name__ == "__main__":
    demo_gui_automation()

Key Concepts to Remember​

1. Multi-State Activation​

Transitions can activate multiple states atomically:

activate_states=['toolbar', 'sidebar', 'content']  # All activated together

2. Multi-Target Pathfinding​

Find paths that reach ALL specified targets:

path = manager.find_path_to(['target1', 'target2', 'target3'])
# Returns optimal path reaching ALL three

3. Phased Execution​

Transitions execute in phases with automatic rollback:

  • VALIDATE: Check preconditions
  • ACTIVATE: Add states to memory (can't fail)
  • INCOMING: Execute entry actions
  • EXIT: Remove old states (only if incoming succeeded)

4. State Groups​

Group related states for atomic operations:

manager.add_group('panels', ['search', 'properties', 'debug'])
manager.activate_group('panels')  # Activate all at once
